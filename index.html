Step1

★経緯

  • 納品物整理のとき、全シートの体裁確認に時間がとられる。
  • インタフェース整理の作業において、いろんな設計書や仕様書を確認しなければならないが、シートのフォーカス位置や拡大率がバラバラで見にくい。

★やりたいこと

  • フォーカス位置を「A1」にする
  • シートの拡大率を「100％」にする
  • 行の高さ、列の幅を揃える
  • フォントを「Meiryo UI」にする（図形の中の文字もできれば。）

★計画

  • 納品物整理においては、上記やりたいことすべてを適応
  • 過去仕様書、設計書においては、体裁が崩れる可能性があるため、以下のみ適応
    • フォーカス位置を「A1」にする
    • シートの拡大率を「100％」にする

        Sub FocusA1_ActiveSheet()
        	Dim wb As Workbook
        	Dim ws As Worksheet
            	Dim shp As Shape
            
        	'==== 設定値 ====
        	Const ROW_H As Double = 15.75
        	Const COL_W As Double = 1.5
        	'==== 適用範囲（必要に応じて変更）====
        	Const COL_FROM As String = "A"
        	Const COL_TO As String = "AX"
        	Const ROW_FROM As Long = 1
        	Const ROW_TO As Long = 200
        	Set wb = ActiveWorkbook
         
        	For Each ws In wb.Worksheets
        		ws.Activate
        		With ActiveWindow
        			.ScrollRow = 1
        			.ScrollColumn = 1
        			.Zoom = 100
        		End With
        		ws.Range("A1").Select
        		ws.Cells.Font.Name = "Meiryo UI"
        
        		 'ws.Columns(COL_FROM & ":" & COL_TO).ColumnWidth = COL_W
        		ws.Cells.ColumnWidth = COL_W
        		 'ws.Rows(ROW_FROM & ":" & ROW_TO).RowHeight = ROW_H
        		ws.Cells.RowHeight = ROW_H
        
        	For Each shp In ws.Shapes
        		If shp.TextFrame2.HasText Then
        			shp.TextFrame2.TextRange.Font.Name = "Meiryo UI"
        
        		ActiveSheet.Shapes.SelectAll
        		With Selection.ShapeRange.TextFrame2.TextRange.Font
        			.NameComplexScript = "Meiryo UI"
        			.NameFarEast = "Meiryo UI"
        			.Name = "Meiryo UI"
        		End With
        
        		End If
        	Next shp
        
            Next ws
        End Sub
